    \begin{exercise}

      Graficar la masa semi-empírica \(M(Z, A)\) para \(A = 14\) como función de \(Z\).
    \end{exercise}

    % Problema 3
    \begin{exercise}
      Demuestre que el isóbaro de menor masa es el \ch{^{14}N}, que además es el único estable. Todos los demás isóbaros decaen por emisión \(\beta\).
      \textbf{Hint: Derivar y buscar \(Z\) mínimo}

      \begin{equation}
        \left[\pdv{M(Z, A)}{Z}\right]_{A = \text{cte} = 14} = 0.
      \end{equation}
    \end{exercise}

    % Problema 4
    \begin{exercise}
      A partir de \(\rho(r)\), ecuación \cref{eq:densitydistribution}, y  los datos dados en \cref{eq:density0} calcular y graficar la distribución de densidad de nucleones en \ch{^{238}U}.

      \begin{equation}
        \rho(r) = \dfrac{\rho_{0}}{1 + \exp[\frac{r - R}{a}]}.
        \label{eq:densitydistribution}
      \end{equation}

      \begin{align}
        \rho_{0} &\simeq \SI[per-mode = symbol]{1.65e44}{\nucleones\per\cubic\metre} = \SI[per-mode = symbol]{0.165}{\nucleones\per\femto\metre\cubed},\label{eq:density0}\\
        R &\simeq \SI[parse-numbers=false]{1.07 A^{\frac{1}{3}}}{\femto\metre},\nonumber\\
        a &\simeq \SI{0.55}{\femto\metre}.\nonumber
      \end{align}
    \end{exercise}

    % Problema 5
    \begin{exercise}
      Demuestra que la emisión alfa de \ch{^{226}_{88}Ra -> ^{222}_{86}Rn + ^{4}_{2}He} es enérgicamente posible y calcule el valor de la energía liberada \(Q\) en el proceso. Para el valor de las masas puede apoyarse del apéndice C del libro Introductory Nuclear Physics. Krane, Kenneth S. John Wiley and Sons. 1987 o de cualquier referencia dada en la conferencia 1.
    \end{exercise}
